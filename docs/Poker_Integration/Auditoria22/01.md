## 7. Queries de Verificacao

```sql
-- 1. Verificar duplicatas em transactions
SELECT player_id, week_period_id, amount, type, COUNT(*) as count
FROM poker_chip_transactions
GROUP BY player_id, week_period_id, amount, type
HAVING COUNT(*) > 1;

-- 2. Verificar duplicatas em settlements
SELECT player_id, week_period_id, gross_amount, COUNT(*) as count
FROM poker_settlements
GROUP BY player_id, week_period_id, gross_amount
HAVING COUNT(*) > 1;

-- 3. Verificar settlements com rakeback = 0 (bug)
SELECT id, player_id, gross_amount, rakeback_amount, net_amount
FROM poker_settlements
WHERE rakeback_amount = 0 AND gross_amount > 0;

-- 4. Verificar consistencia chip_balance vs transactions
SELECT
  p.id,
  p.chip_balance as stored_balance,
  COALESCE(SUM(
    CASE
      WHEN t.type IN ('chip_in','credit_received','bonus','adjustment_positive',
                      'transfer_in','rake_refund','jackpot') THEN t.amount
      ELSE -t.amount
    END
  ), 0) as calculated_balance
FROM poker_players p
LEFT JOIN poker_chip_transactions t ON t.player_id = p.id
GROUP BY p.id
HAVING p.chip_balance != calculated_balance;

-- 5. Verificar transacoes orfas (sem player)
SELECT COUNT(*) as orphan_count
FROM poker_chip_transactions
WHERE player_id IS NULL;
```
