## Exemplos de Uso

Explore diferentes formas de integrar e utilizar o Mid Poker.

### Sincronização de Transações

```typescript
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.SUPABASE_URL,
  process.env.SUPABASE_KEY
);

// Buscar transações recentes
const { data: transactions } = await supabase
  .from('transactions')
  .select('*')
  .eq('team_id', teamId)
  .order('date', { ascending: false })
  .limit(50);
```

### Gerar Relatório de Lucro/Perda

```typescript
// Via tRPC
const report = await trpc.reports.profit.query({
  from: '2024-01-01',
  to: '2024-12-31',
  currency: 'BRL',
  revenueType: 'net'
});

console.log('Lucro total:', report.summary.profit);
console.log('Receita:', report.summary.revenue);
console.log('Despesas:', report.summary.expenses);
```

### Criar Fatura

```typescript
const invoice = await trpc.invoice.create.mutate({
  customerId: 'customer-uuid',
  dueDate: '2024-02-01',
  lineItems: [
    {
      description: 'Stake 50/50 - Janeiro 2024',
      quantity: 1,
      price: 5000,
    }
  ],
  currency: 'BRL'
});
```

### Webhook de Transação

```typescript
// Endpoint para receber webhooks
export async function POST(req: Request) {
  const payload = await req.json();

  if (payload.type === 'transaction.created') {
    // Nova transação recebida
    const { amount, name, category } = payload.data;

    // Enviar notificação, atualizar dashboard, etc.
    await sendSlackNotification({
      text: `Nova transação: ${name} - R$ ${amount}`
    });
  }

  return Response.json({ received: true });
}
```
